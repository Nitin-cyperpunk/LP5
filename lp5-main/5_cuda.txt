#include<stdio.h>
#include<cuda.h>

#define N 3

// -------- Vector Addition --------
__global__ void vectorAdd(int *a, int *b, int *c)
{
    int i = threadIdx.x;
    c[i] = a[i] + b[i];
}

// -------- Matrix Multiplication --------
__global__ void matrixMul(int a[N][N], int b[N][N], int c[N][N])
{
    int row = threadIdx.x;
    int col = threadIdx.y;

    int sum = 0;

    for(int k = 0; k < N; k++)
    {
        sum += a[row][k] * b[k][col];
    }

    c[row][col] = sum;
}

int main()
{
    // -------- Vector Addition --------
    int a[N] = {1,2,3};
    int b[N] = {4,5,6};
    int c[N];

    int *d_a, *d_b, *d_c;

    cudaMalloc(&d_a, N*sizeof(int));
    cudaMalloc(&d_b, N*sizeof(int));
    cudaMalloc(&d_c, N*sizeof(int));

    cudaMemcpy(d_a, a, N*sizeof(int), cudaMemcpyHostToDevice);
    cudaMemcpy(d_b, b, N*sizeof(int), cudaMemcpyHostToDevice);

    vectorAdd<<<1,N>>>(d_a,d_b,d_c);

    cudaMemcpy(c, d_c, N*sizeof(int), cudaMemcpyDeviceToHost);

    printf("Vector Addition:\n");

    for(int i=0;i<N;i++)
        printf("%d ", c[i]);

    // -------- Matrix Multiplication --------
    int A[N][N] = {{1,2,3},{4,5,6},{7,8,9}};
    int B[N][N] = {{1,0,0},{0,1,0},{0,0,1}};
    int C[N][N];

    int (*d_A)[N], (*d_B)[N], (*d_C)[N];

    cudaMalloc(&d_A, sizeof(A));
    cudaMalloc(&d_B, sizeof(B));
    cudaMalloc(&d_C, sizeof(C));

    cudaMemcpy(d_A, A, sizeof(A), cudaMemcpyHostToDevice);
    cudaMemcpy(d_B, B, sizeof(B), cudaMemcpyHostToDevice);

    dim3 threads(N,N);

    matrixMul<<<1,threads>>>(d_A,d_B,d_C);

    cudaMemcpy(C, d_C, sizeof(C), cudaMemcpyDeviceToHost);

    printf("\n\nMatrix Multiplication:\n");

    for(int i=0;i<N;i++)
    {
        for(int j=0;j<N;j++)
        {
            printf("%d ", C[i][j]);
        }
        printf("\n");
    }

    return 0;
}
