import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.preprocessing import MinMaxScaler
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense, LSTM, Dropout

# CELL 2 - Load Dataset
# Downloads Google stock data automatically (no CSV upload needed)
import yfinance as yf
data = yf.download('GOOG', start='2015-01-01', end='2023-12-31')

print("Dataset Shape:", data.shape)
print(data.head())

# Use 'Open' price for prediction (same as PDF)
dataset = data['Open'].values.reshape(-1, 1)

# CELL 3 - Preprocess - Scale Data
scaler  = MinMaxScaler(feature_range=(0, 1))
dataset = scaler.fit_transform(dataset)

# Split 80% train, 20% test
training_data_len = int(len(dataset) * 0.8)
training_data     = dataset[:training_data_len]
testing_data      = dataset[training_data_len:]

print("Training samples:", len(training_data))
print("Testing  samples:", len(testing_data))

# CELL 4 - Create Dataset with Time Steps
def create_dataset(dataset, time_step=60):
    X, Y = [], []
    for i in range(len(dataset) - time_step - 1):
        X.append(dataset[i:(i + time_step), 0])
        Y.append(dataset[i + time_step, 0])
    return np.array(X), np.array(Y)

time_step = 60
X_train, Y_train = create_dataset(training_data, time_step)
X_test,  Y_test  = create_dataset(testing_data,  time_step)

# Reshape for LSTM input: (samples, time_steps, features)
X_train = X_train.reshape(X_train.shape[0], X_train.shape[1], 1)
X_test  = X_test.reshape(X_test.shape[0],   X_test.shape[1],  1)

print("X_train shape:", X_train.shape)
print("X_test  shape:", X_test.shape)

# CELL 5 - Build the RNN Model (from PDF)
model = Sequential()
model.add(LSTM(units=50, return_sequences=True,
               input_shape=(X_train.shape[1], 1)))
model.add(Dropout(0.2))

model.add(LSTM(units=50, return_sequences=True))
model.add(Dropout(0.2))

model.add(LSTM(units=50))
model.add(Dropout(0.2))

model.add(Dense(units=1))   # Output: next day's stock price

model.summary()

# CELL 6 - Compile the Model
model.compile(optimizer='adam', loss='mean_squared_error')

# CELL 7 - Train the Model
history = model.fit(X_train, Y_train,
                    epochs=20,
                    batch_size=32,
                    validation_split=0.1)

# CELL 8 - Evaluate
train_loss = model.evaluate(X_train, Y_train, verbose=0)
test_loss  = model.evaluate(X_test,  Y_test,  verbose=0)
print(f"\nTrain Loss (MSE): {train_loss:.6f}")
print(f"Test  Loss (MSE): {test_loss:.6f}")

# CELL 9 - Plot Training Loss
plt.figure(figsize=(8, 4))
plt.plot(history.history['loss'],     label='Train Loss')
plt.plot(history.history['val_loss'], label='Val Loss')
plt.title('Model Loss over Epochs')
plt.xlabel('Epochs')
plt.ylabel('Loss (MSE)')
plt.legend()
plt.tight_layout()
plt.show()

# CELL 10 - Predict & Inverse Transform
train_predict = scaler.inverse_transform(model.predict(X_train))
test_predict  = scaler.inverse_transform(model.predict(X_test))
Y_train_inv   = scaler.inverse_transform(Y_train.reshape(-1, 1))
Y_test_inv    = scaler.inverse_transform(Y_test.reshape(-1, 1))

# CELL 11 - Plot Actual vs Predicted
plt.figure(figsize=(12, 5))
plt.plot(data['Open'].values, label='Actual Price', color='blue')

# Shift predictions to correct position on x-axis
train_plot = np.empty_like(dataset)
train_plot[:] = np.nan
train_plot[time_step:len(train_predict)+time_step] = train_predict

test_plot = np.empty_like(dataset)
test_plot[:] = np.nan
test_plot[training_data_len + time_step:
          training_data_len + time_step + len(test_predict)] = test_predict

plt.plot(scaler.inverse_transform(train_plot), label='Train Predict', color='orange')
plt.plot(scaler.inverse_transform(test_plot),  label='Test Predict',  color='green')
plt.title('Google Stock Price - Actual vs Predicted')
plt.xlabel('Days')
plt.ylabel('Stock Price (USD)')
plt.legend()
plt.tight_layout()
plt.show()

# CELL 12 - Predict Next Day Price
last_60_days  = dataset[-60:]
last_60_days  = last_60_days.reshape(1, 60, 1)
next_day_pred = model.predict(last_60_days)
next_day_price = scaler.inverse_transform(next_day_pred)
print(f"\nPredicted Next Day Opening Price: ${next_day_price[0][0]:.2f}")
